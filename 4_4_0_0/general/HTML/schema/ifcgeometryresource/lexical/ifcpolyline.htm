ENTITY IfcPolyline
 SUBTYPE OF (IfcBoundedCurve);
  Points : LIST [2:?] OF IfcCartesianPoint;
 WHERE
  SameDim : SIZEOF(QUERY(Temp <* Points | Temp.Dim <> Points[1].Dim)) = 0;
END_ENTITY;
